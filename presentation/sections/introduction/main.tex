% --------------------------------------------------------- Fair Signature Exchange Example
\begin{frame}
	\frametitle{Fair Signature Exchange (Example)}

	\begin{itemize}[<+->]
		\setlength\itemsep{1em}
		\item Two parties A(lice) and B(ob)
		\item Want to sign an item (e.g. a contract)\vspace*{.5cm}
		\item[]
			\begin{center}			
				\begin{tikzpicture}[node distance=2cm,auto,>=stealth']
					\visible<3->{\node[] (bob) {Bob};}
					\visible<3->{\node[left = of bob] (alice) {Alice};}
					\visible<3->{\node[below of=bob, node distance=3cm] (bobground) {};}
					\visible<3->{\node[below of=alice, node distance=3cm] (aliceground) {};}
					
					% vertical lines
					\visible<3->{\draw (alice) -- (aliceground);}
					\visible<3->{\draw (bob) -- (bobground);}

					% signs contract -> receives signed contract
					\visible<4->{\draw[->, dashed] ($(alice)!0.2!(aliceground)$) -- node[above, scale=0.6, sloped]{signs contract} ($(bob)!0.25!(bobground)$);}
					\visible<5->{\draw[->, strike thru arrow, dashed] ($(bob)!0.5!(bobground)$) -- ($(alice)!0.6!(aliceground)$);}
				\end{tikzpicture}
			\end{center}
	\end{itemize}
\end{frame}
